// ════════════════════════════════════════════════════════
// Regras de Segurança do Firestore — Painel ML Logística
//
// COMO APLICAR:
// 1. Acesse https://console.firebase.google.com
// 2. Abra o projeto "painel-ml-logistica"
// 3. Menu lateral → Build → Firestore Database → aba "Regras"
// 4. Apague o conteúdo atual e cole o conteúdo abaixo
// 5. Clique em "Publicar"
//
// O QUE ISSO FAZ:
// Qualquer pessoa SEM login não consegue ler nem escrever nada.
// Qualquer colega COM login (criado pela tela de login do painel)
// pode ler e escrever relatórios e a configuração de meta.
// ════════════════════════════════════════════════════════

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // Relatórios diários — qualquer colega autenticado pode
    // criar, ler, atualizar ou apagar (uso de equipe, sem
    // permissões por pessoa nesta primeira versão).
    match /relatorios/{reportId} {
      allow read, write: if request.auth != null;
    }

    // Configuração compartilhada (ex: meta de OOT/SLA).
    match /config/{document} {
      allow read, write: if request.auth != null;
    }

    // Bloqueia qualquer outra coleção por padrão.
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
